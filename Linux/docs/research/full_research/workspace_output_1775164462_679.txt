        result.error += ")";
        return false;
    }
    
    int status;
    waitpid(tid, &status, 0);
    printf("[+] Attached to thread %u\n", tid);
    
    // Get registers
    struct user_regs_struct regs, origRegs;
    if (ptrace(PTRACE_GETREGS, tid, nullptr, &regs) == -1) {
        result.error = "Failed to get registers";
        ptrace(PTRACE_DETACH, tid, nullptr, nullptr);
        return false;
    }
    
    origRegs = regs;
    
    // Find libc in target process
    auto libcMod = GetModule(pid, "libc.so");
    if (!libcMod) {
        // Try libc without version
        auto modules = GetModules(pid);
        for (const auto& mod : modules) {
            if (mod.name.find("libc-") == 0 || mod.name.find("libc.") == 0) {
                libcMod = mod;
                break;
            }
        }
    }
    
    if (!libcMod) {
        result.error = "Failed to find libc in target process";
        ptrace(PTRACE_DETACH, tid, nullptr, nullptr);
        return false;
    }
    
    printf("[+] Found libc at 0x%lX\n", libcMod->base);
    
    // Allocate memory for library path
    size_t pathSize = libraryPath.size() + 1;
    uintptr_t pathMem = AllocateRemoteMemory(pid, pathSize, PROT_READ | PROT_WRITE);
    
    if (pathMem == 0 || pathMem == (uintptr_t)-1) {
        result.error = "Failed to allocate memory for library path";
        ptrace(PTRACE_DETACH, tid, nullptr, nullptr);
        return false;
    }
    
    printf("[+] Allocated path memory at 0x%lX\n", pathMem);
    
    // Write library path
    if (!WriteProcessMemory(pid, pathMem, libraryPath.c_str(), pathSize)) {
        result.error = "Failed to write library path";
        ptrace(PTRACE_DETACH, tid, nullptr, nullptr);
        return false;
    }
    
    // For a complete implementation, we need to:
    // 1. Find dlopen or __libc_dlopen_mode in libc
    // 2. Call it with the library path
    // This requires parsing libc's symbol table
    
    // For Wine processes, we might need to use different approach
    // Wine has its own DLL loading mechanism
    
    printf("[*] Library path written to target process\n");
    printf("[*] Note: Full dlopen call requires symbol resolution\n");
    
    // Restore registers
    ptrace(PTRACE_SETREGS, tid, nullptr, &origRegs);
    
    // Detach
    ptrace(PTRACE_DETACH, tid, nullptr, nullptr);
    
    // For Wine/Proton, suggest alternative method
    if (IsWineProcess(pid)) {
        result.warnings.push_back("Target is Wine/Proton process - consider using LD_PRELOAD method");
    }
    
    result.error = "Ptrace injection incomplete - requires symbol resolution for dlopen";
    return false;
}
